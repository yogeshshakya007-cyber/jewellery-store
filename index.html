async function loadProducts() { const container = document.getElementById("product-list"); container.innerHTML = `

Loading collections...

`; try { if (!window.supabase) { throw new Error("Supabase library not loaded."); } const supabase = window.supabase.createClient( supabaseUrl, supabaseKey ); const { data, error } = await supabase .from("products") .select("*") .order("id", { ascending: false }); if (error) { console.error(error); throw error; } if (!data || data.length === 0) { container.innerHTML = `
No Products Found
`; return; } container.innerHTML = ""; data.forEach(product => { container.innerHTML += `

${product.name}

₹${product.price}

`; }); } catch(err){ console.error("LOAD ERROR",err); container.innerHTML=`

Failed to load products

${err.message}

`; } }
